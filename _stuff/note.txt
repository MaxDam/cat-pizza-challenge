def check_what_is_empty(pizza_order):
    ask_for = []
    for field, value in pizza_order.dict().items():
        if value in [None, "", 0]:
            print(f"Field '{field}' is empty.")
            ask_for.append(f'{field}')
    return ask_for
	
	
def add_non_empty_details(current_order: PizzaOrder, new_order: PersonalDetails):
    non_empty_details = {k: v for k, v in new_order.dict().items() if v not in [None, ""]}
    updated_details = current_order.copy(update=non_empty_details)
    return updated_details
